# 處理訂單資料

在頁首輸入處理訂單資料的程式碼

// 將座位陣列序列化以存入資料庫
$seat = serialize($_POST["seat"]);
// 訂單編號為訂購日期8碼+資料ID補0至4位數
$num = date("Ymd", $_GET["d"]).str_pad($pdo->lastInsertId(), 4, "0", STR_PAD_LEFT);
// 寫入資料庫
All("insert into orders values(null, ".$_GET["m"].", ".$_GET["d"].", ".$_GET["t"].", '".$seat."', '".$num."')");

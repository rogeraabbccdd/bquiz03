處理表單

在 api.php 加入處理表單的程式碼

// 單筆刪除
case "delord":
    All("delete from orders where id = '".$_GET["id"]."'");
    lo("admin.php?redo=order");
    break;

// 快速刪除
case "fastdel":
    // 依日期
    if($_POST["mode"] == 1)
    {
        $date = strtotime($_POST["date"]);
        All("delete from orders where date = '".$date."'");
    }
    // 依電影
    elseif($_POST["mode"] == 2)
    {
        All("delete from orders where movie = '".$_POST["movie"]."'");
    }
    lo("admin.php?redo=order");
    break;
